class InspectRequest(BaseModel):
    subject_id: str = "default"
    query_text: str
    agent_role: str = "assistant"
    session_context: dict[str, Any] = {}
    limit: int = 12


def _bundle_to_dict(bundle) -> dict:
    """Serialize a MemoryBundle to a JSON-safe dict."""
    def items_to_list(items) -> list[dict]:
        return [
            {
                "memory_id": i.memory_id,
                "memory_type": i.memory_type,
                "title": i.title,
                "content": i.content,
                "origin_type": i.origin_type,
                "review_status": i.review_status,
                "confidence": round(i.confidence, 3),
                "salience": round(i.salience, 3),
            }
            for i in items
        ]
    return {
        "constraints":         items_to_list(bundle.constraints),
        "priorities":          items_to_list(bundle.priorities),
        "open_loops":          items_to_list(bundle.open_loops),
        "recent_decisions":    items_to_list(bundle.recent_decisions),
        "relationship_context": items_to_list(bundle.relationship_context),
        "contradictions":      items_to_list(bundle.contradictions),
        "interaction_rules":   items_to_list(bundle.interaction_rules),
        "session_relevant":    items_to_list(bundle.session_relevant),
        "trace":               bundle.trace,
    }


@app.post("/api/memory/provider/inspect")
def api_memory_provider_inspect(body: InspectRequest) -> dict:
    try:
        provider = _load_provider()
        bundle = provider.get_operational_memory(
            subject_id=body.subject_id,
            query_text=body.query_text,
            agent_role=body.agent_role,
            session_context=body.session_context,
            limit=body.limit,

// ── Context Inspector ─────────────────────────────────────────────────────
async function loadCtxInspector() {
  const s = await get('/api/memory/provider/status');
  const provEl = document.getElementById('ctx-provider-status');
  const cls = s.healthy ? 'badge-success' : 'badge-danger';
  provEl.innerHTML = `<span class="badge ${cls}">${s.provider}</span>
    <span style="font-size:12px;color:var(--muted-fg);margin-left:10px">${s.detail || ''}</span>`;
}

async function runInspect() {
  const query   = document.getElementById('ctx-query').value.trim();
  const role    = document.getElementById('ctx-role').value;
  const subject = document.getElementById('ctx-subject').value.trim() || 'default';
  const limit   = parseInt(document.getElementById('ctx-limit').value, 10) || 12;
  const el      = document.getElementById('ctx-bundle-result');
  if (!query) { el.innerHTML = '<p style="color:var(--warn);font-size:13px">Inserisci un query text.</p>'; return; }
  el.innerHTML = '<p style="color:var(--muted-fg);font-size:13px">Caricamento…</p>';
  try {
    const bundle = await post('/api/memory/provider/inspect', { subject_id: subject, query_text: query, agent_role: role, session_context: {}, limit });
    el.innerHTML = renderBundle(bundle);
  } catch(err) { el.innerHTML = `<p style="color:var(--danger);font-size:13px">Errore: ${err.message}</p>`; }
}

function renderBundle(bundle) {
  const LABELS = {
    constraints:          { label: 'Constraints',          icon: '◈', cls: 'badge-danger'  },
    priorities:           { label: 'Priorities',           icon: '◉', cls: 'badge-warn'    },
    open_loops:           { label: 'Open Loops',           icon: '◎', cls: 'badge-accent'  },
    recent_decisions:     { label: 'Recent Decisions',     icon: '◑', cls: 'badge-blue'    },
    relationship_context: { label: 'Relationship Context', icon: '◌', cls: 'badge-purple'  },
    contradictions:       { label: 'Contradictions',       icon: '⚡', cls: 'badge-danger'  },
    interaction_rules:    { label: 'Interaction Rules',    icon: '◐', cls: 'badge-muted'   },
    session_relevant:     { label: 'Session Relevant',     icon: '⬡', cls: 'badge-muted'   },
  };
  const keys = Object.keys(LABELS);
  const totalItems = keys.reduce((n, k) => n + (bundle[k]?.length || 0), 0);
  if (totalItems === 0) {
    return emptyState('◎', 'Bundle vuoto', 'Provider null o nessun item sopra soglia confidence');
  }
  return keys.map(k => {
    const items = bundle[k] || [];
    if (!items.length) return '';
    const { label, icon, cls } = LABELS[k];
    return `<div class="card" style="margin-bottom:12px">
      <div class="card-title">${icon} ${label} <span class="badge ${cls}" style="margin-left:6px">${items.length}</span></div>
      ${items.map(i => `
        <div class="mem-card" style="margin-bottom:8px">
          <div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:4px">
            <span style="font-weight:600;font-size:13px">${i.title}</span>
            <div style="display:flex;gap:4px;flex-shrink:0">
              <span class="badge badge-muted">${i.memory_type}</span>
              <span class="badge badge-muted" style="font-family:var(--font-mono)">${(i.confidence*100).toFixed(0)}%</span>
              <span class="badge badge-muted">${i.origin_type}</span>
            </div>
          </div>
          <p class="mem-content">${i.content}</p>
          <div class="mem-meta">id: ${i.memory_id}</div>
        </div>`).join('')}
    </div>`;
  }).join('') + renderTrace(bundle.trace);
}

function renderTrace(trace) {
  if (!trace?.length) return '';
  const top = trace.slice(0, 10);
  const rows = top.map(t => `
    <tr>
      <td class="mono">${t.memory_id || t.source || '-'}</td>
      <td>${t.memory_type || t.step || '-'}</td>
      <td class="mono">${typeof t.score === 'number' ? t.score.toFixed(3) : '-'}</td>
      <td>${t.review_status || '-'}</td>
    </tr>`).join('');
  return `<div class="card" style="margin-top:12px">
    <div class="card-title">Retrieval Trace <span class="badge badge-muted" style="margin-left:6px">${trace.length} candidati</span></div>
    <div class="table-wrap"><table>
      <thead><tr><th>ID</th><th>Type / Step</th><th>Score</th><th>Review</th></tr></thead>
      <tbody>${rows}</tbody>
    </table></div>
  </div>`;
}

# ── Amber ─────────────────────────────────────────────────────────────────────

def _amber_db():
    """Return an initialized AmberDB or None if amber is unavailable.

    OSS: requires AMBER_DATA_DIR env var; returns None if not set (no amber in demo).
    Local private webui: overrides this function with a hardcoded path.
    """
    data_dir = _os.environ.get("AMBER_DATA_DIR", "")
    if not data_dir:
        return None
    try:
        import sys as _sys
        # Try amber package from PYTHONPATH or sibling Amber-Soulkiller repo
        try:
            from amber.db import AmberDB
        except ImportError:
            _amber_root = Path(__file__).resolve().parents[3] / "Amber-Soulkiller"
            if _amber_root.exists():
                _sys.path.insert(0, str(_amber_root))
            from amber.db import AmberDB
        from pathlib import Path as _Path
        db = AmberDB(db_path=_Path(data_dir) / "amber-operational.sqlite")
        db.initialize()
        return db
    except Exception:
        return None


@app.get("/api/amber/status")
def api_amber_status() -> dict:
    db = _amber_db()
    if db is None:
        return {"available": False, "healthy": False, "detail": "amber package not installed"}
    try:
        from amber.provider import AmberMemoryProvider
        provider = AmberMemoryProvider(db=db)
        status = provider.health_check()
        row = db.conn.execute(
            "SELECT COUNT(*) as n FROM operational_memory_items WHERE status='active'"
        ).fetchone()
        active_items = dict(row)["n"] if row else 0
        return {
            "available": True,
            "healthy": status.healthy,
            "detail": status.detail,
            "active_items": active_items,
        }
    except Exception as exc:
        return {"available": True, "healthy": False, "detail": str(exc)}
    finally:
        db.close()
```

@app.get("/api/amber/items")
def api_amber_items(
    subject_id: str = "",
    review_status: str = "",
    memory_type: str = "",
    limit: int = 100,
) -> list[dict]:
    db = _amber_db()
    if db is None:
        return []
    try:
        conditions = ["status = 'active'"]
        params: list[Any] = []
        if subject_id:
            conditions.append("subject_id = ?")
            params.append(subject_id)
        if review_status:
            conditions.append("review_status = ?")
            params.append(review_status)
        if memory_type:
            conditions.append("memory_type = ?")
            params.append(memory_type)
        where = " AND ".join(conditions)
        rows = db.conn.execute(
            f"""
            SELECT id, subject_id, memory_type, title, content,
                   origin_type, review_status, confidence, salience,
                   last_seen_at, created_at
            FROM operational_memory_items
            WHERE {where}
            ORDER BY confidence DESC, last_seen_at DESC
            LIMIT ?
            """,
            (*params, limit),
        ).fetchall()
        return [dict(r) for r in rows]
    finally:
        db.close()
```

@app.post("/api/amber/items/{memory_id}/review")
def api_amber_review(memory_id: str, body: AmberReviewRequest) -> dict:
    if body.action not in _VALID_REVIEW_ACTIONS:
        raise HTTPException(
            status_code=422,
            detail=f"Invalid action. Must be one of: {sorted(_VALID_REVIEW_ACTIONS)}",
        )
    db = _amber_db()

        raise HTTPException(status_code=503, detail="amber not available")
    try:
        row = db.conn.execute(
            "SELECT id FROM operational_memory_items WHERE id = ?", (memory_id,)
        ).fetchone()
        if not row:
            raise HTTPException(status_code=404, detail="memory item not found")
        import sys as _sys
        _amber_root = Path(__file__).resolve().parents[3] / "Amber-Soulkiller"
        if _amber_root.exists() and str(_amber_root) not in _sys.path:
            _sys.path.insert(0, str(_amber_root))
        from amber.provider import AmberMemoryProvider
        provider = AmberMemoryProvider(db=db)
        provider.review_memory_item(memory_id, body.action, body.note)
        return {"ok": True, "memory_id": memory_id, "action": body.action}
    finally:
        db.close()
